172  inline double logadd(double log_a, double log_b){
173 
174  if (log_a < log_b)
175  { // swap them
176  double tmp = log_a;
177  log_a = log_b;
178  log_b = tmp;
179  } else if (fast_exact_is_equal(log_a, log_b)) {
180  // Special case when log_a == log_b. In particular this works when both
181  // log_a and log_b are (+-) INFINITY: it will return (+-) INFINITY
182  // instead of NaN.
183  return LOG_2 + log_a;
184  }
185  double negative_absolute_difference = log_b - log_a;
186  if (negative_absolute_difference < MINUS_LOG_THRESHOLD)
187  return (log_a);
188  return (log_a + log1p(exp(negative_absolute_difference)));
189 
190  }
