215 
216  std::string config_file = vm["config"].as<std::string>();
217 
218  std::ifstream ifs(config_file.c_str());
219  if (!ifs){
220  elly::utils::log() << "can not open config file: " << config_file << "\n";
221  return 1;
222  }
223 
224  po::variables_map file_vm;
225  po::store(parse_config_file(ifs, desc), file_vm);
226  notify(file_vm);
227 
228  parse_option_fill_in_config(config, file_vm, desc);
229 
230  ifs.close();
231 
232  }
233 
234  elly::utils::log() << ">> Parsing command line options... " << std::endl;
235 
236  int rs_parse_options_post = parse_option_fill_in_config(config, vm, desc);
237 
238  parse_option_post(config);
239 
240  return rs_parse_options_post;
